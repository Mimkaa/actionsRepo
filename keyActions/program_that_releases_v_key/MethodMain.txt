public static void main(String[] args) {
    try {
        java.awt.Robot r = new java.awt.Robot();
        r.delay(500);
        r.keyPress(java.awt.event.KeyEvent.VK_V);
        r.keyRelease(java.awt.event.KeyEvent.VK_V);
        System.out.println("Pressed V");
    } catch (Exception e) {
        e.printStackTrace();
    }
}
