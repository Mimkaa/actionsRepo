public static void main(String[] args){
    try{
        System.out.println("Attempting to press key '7' via java.awt.Robot...");
        java.awt.Robot r = new java.awt.Robot();
        r.setAutoDelay(50);
        r.delay(300);
        r.keyPress(java.awt.event.KeyEvent.VK_7);
        r.keyRelease(java.awt.event.KeyEvent.VK_7);
        System.out.println("Pressed '7'.");
    }catch(java.awt.AWTException e){
        System.out.println("Failed to create Robot (AWTException): "+e);
        e.printStackTrace();
    }catch(Throwable t){
        System.out.println("Unexpected error: "+t);
        t.printStackTrace();
    }

    try{
        java.nio.file.Files.write(java.nio.file.Paths.get(".ready"), "done".getBytes(java.nio.charset.StandardCharsets.UTF_8));
        System.out.println("Created .ready");
    }catch(Throwable t){
        System.out.println("Failed to create .ready: "+t);
    }
}
