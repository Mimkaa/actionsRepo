public static void main(String[] args) {
    System.out.println("Pressing 5 key...");
    try {
        java.awt.Robot r = new java.awt.Robot();
        r.setAutoDelay(50);
        r.keyPress(java.awt.event.KeyEvent.VK_5);
        r.keyRelease(java.awt.event.KeyEvent.VK_5);
        System.out.println("Done.");
    } catch (java.awt.AWTException e) {
        e.printStackTrace();
    }

    // Terminate pipeline loop by creating .ready without using CreateTextFile
    try {
        java.nio.file.Files.write(
            java.nio.file.Paths.get(".ready"),
            "ready\n".getBytes(java.nio.charset.StandardCharsets.UTF_8)
        );
        System.out.println("Created .ready");
    } catch (Exception e) {
        e.printStackTrace();
    }
}
