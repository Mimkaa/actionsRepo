public static void main(String[] args){
    System.out.println("Pressing '/' key in 1 second...");
    try{
        Thread.sleep(1000);
        java.awt.Robot r=new java.awt.Robot();
        r.keyPress(java.awt.event.KeyEvent.VK_SLASH);
        r.keyRelease(java.awt.event.KeyEvent.VK_SLASH);
        System.out.println("Done.");
    }catch(Throwable t){
        t.printStackTrace();
    }

    // Terminate the STECHEN loop by creating a real file named .ready (not via CreateTextFile)
    try{
        java.nio.file.Path p=java.nio.file.Paths.get(".ready");
        if(!java.nio.file.Files.exists(p)){
            java.nio.file.Files.createFile(p);
        }
    }catch(Throwable t){
        t.printStackTrace();
    }
}
