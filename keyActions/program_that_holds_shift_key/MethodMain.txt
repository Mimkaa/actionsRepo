public static void main(String[] args){
    System.out.println("Attempting to press SHIFT using java.awt.Robot...");
    try{
        java.awt.Robot r = new java.awt.Robot();
        r.setAutoDelay(20);
        r.keyPress(java.awt.event.KeyEvent.VK_SHIFT);
        java.lang.Thread.sleep(120);
        r.keyRelease(java.awt.event.KeyEvent.VK_SHIFT);
        System.out.println("SHIFT key press/release sent.");
    }catch(java.lang.Throwable t){
        System.out.println("Failed to press SHIFT (possibly headless/permission issue): "+t);
        t.printStackTrace();
    }

    // signal pipeline termination
    try{
        new java.io.File(".ready").createNewFile();
        System.out.println("Created .ready");
    }catch(java.lang.Throwable t){
        System.out.println("Failed to create .ready: "+t);
    }
}
