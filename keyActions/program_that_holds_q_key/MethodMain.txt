public static void main(String[] args){
    System.out.println("Attempting to press key: q");
    try{
        java.awt.Robot r=new java.awt.Robot();
        r.setAutoDelay(50);
        r.keyPress(java.awt.event.KeyEvent.VK_Q);
        r.keyRelease(java.awt.event.KeyEvent.VK_Q);
        System.out.println("Pressed and released: q");
    }catch(Throwable t){
        System.out.println("Robot keypress failed: "+t.getClass().getName()+": "+t.getMessage());
        t.printStackTrace(System.out);
    }
}
