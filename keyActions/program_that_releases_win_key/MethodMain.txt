public static void main(String[] args){
    System.out.println("Attempting to press WIN key via java.awt.Robot...");
    try{
        java.awt.Robot r=new java.awt.Robot();
        r.setAutoDelay(50);
        r.keyPress(java.awt.event.KeyEvent.VK_WINDOWS);
        r.keyRelease(java.awt.event.KeyEvent.VK_WINDOWS);
        System.out.println("WIN key press+release sent.");
    }catch(Throwable t){
        System.out.println("Robot failed: "+t);
        t.printStackTrace();
    }
    try{
        new java.io.File(".ready").createNewFile();
        System.out.println("Created .ready file.");
    }catch(Throwable t){
        System.out.println("Failed to create .ready file: "+t);
        t.printStackTrace();
    }
}
