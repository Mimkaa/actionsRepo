public static void main(String[] args) {
    System.out.println("Pressing F2 in 2 seconds...");
    try {
        Thread.sleep(2000);
        java.awt.Robot r = new java.awt.Robot();
        r.setAutoDelay(50);
        r.keyPress(java.awt.event.KeyEvent.VK_F2);
        r.keyRelease(java.awt.event.KeyEvent.VK_F2);
        System.out.println("F2 pressed.");
    } catch (InterruptedException ie) {
        Thread.currentThread().interrupt();
        System.out.println("Interrupted; aborting.");
    } catch (java.awt.AWTException awte) {
        System.out.println("Failed to create Robot: " + awte);
    }
}
