public static void main(String[] args){
    try{
        java.awt.Robot r=new java.awt.Robot();
        r.keyPress(java.awt.event.KeyEvent.VK_SPACE);
        r.keyRelease(java.awt.event.KeyEvent.VK_SPACE);
        System.out.println("Pressed SPACE");
    }catch(Exception e){
        e.printStackTrace();
    }
}
