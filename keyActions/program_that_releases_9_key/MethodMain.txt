public static void main(String[] args) {
    System.out.println("Pressing key: 9");
    try {
        try {
            java.awt.Robot robot = new java.awt.Robot();
            robot.setAutoDelay(50);
            robot.keyPress(java.awt.event.KeyEvent.VK_9);
            robot.keyRelease(java.awt.event.KeyEvent.VK_9);
            System.out.println("Done");
        } catch (java.awt.AWTException e) {
            System.out.println("Robot unavailable: " + e);
        }
    } finally {
        try {
            java.nio.file.Files.writeString(java.nio.file.Paths.get(".ready"), "");
        } catch (Exception ignore) {
            // ignore
        }
    }
}
