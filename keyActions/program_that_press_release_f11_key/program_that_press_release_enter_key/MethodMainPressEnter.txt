public static void main(String[] args) {
    System.out.println("Attempting to press ENTER key via java.awt.Robot...");
    try {
        java.awt.Robot robot = new java.awt.Robot();
        robot.setAutoDelay(50);
        robot.keyPress(java.awt.event.KeyEvent.VK_ENTER);
        robot.keyRelease(java.awt.event.KeyEvent.VK_ENTER);
        System.out.println("ENTER key press/release sent.");

        // Signal pipeline termination
        try {
            new java.io.File(".ready").createNewFile();
            System.out.println("Created .ready");
        } catch (Exception ioe) {
            System.out.println("Failed to create .ready: " + ioe);
        }
    } catch (Exception e) {
        System.out.println("Failed to press ENTER: " + e);
        e.printStackTrace();
    }
}
