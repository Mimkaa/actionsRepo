public static void main(String[] args){
    try{
        System.out.println("Pressing F10...");
        java.awt.Robot r = new java.awt.Robot();
        r.setAutoDelay(50);
        r.keyPress(java.awt.event.KeyEvent.VK_F10);
        r.keyRelease(java.awt.event.KeyEvent.VK_F10);
        System.out.println("Done.");

        // Terminate STECHEN loop by creating a file named .ready (not via CreateTextFile script)
        java.nio.file.Path p = java.nio.file.Paths.get(".ready");
        if(java.nio.file.Files.notExists(p)){
            java.nio.file.Files.write(p, new byte[0]);
        }
    }catch(Throwable t){
        t.printStackTrace();
    }
}
