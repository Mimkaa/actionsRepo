public static void main(String[] args) {
    System.out.println("Attempting to press ';' key via java.awt.Robot...");
    if (java.awt.GraphicsEnvironment.isHeadless()) {
        System.out.println("Headless environment detected; cannot use Robot to press keys.");
        return;
    }
    try {
        java.awt.Robot r = new java.awt.Robot();
        r.setAutoDelay(50);
        r.delay(500);
        r.keyPress(java.awt.event.KeyEvent.VK_SEMICOLON);
        r.keyRelease(java.awt.event.KeyEvent.VK_SEMICOLON);
        System.out.println("Pressed ';' key.");
    } catch (java.awt.AWTException e) {
        System.out.println("Failed to create Robot: " + e);
    }
}
