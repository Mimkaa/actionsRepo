public static void main(String[] args){
  try{
    java.awt.Robot r=new java.awt.Robot();
    r.delay(500);
    r.keyPress(java.awt.event.KeyEvent.VK_G);
    r.keyRelease(java.awt.event.KeyEvent.VK_G);
    System.out.println("Pressed g");

  }catch(Exception e){
    e.printStackTrace();
  }
