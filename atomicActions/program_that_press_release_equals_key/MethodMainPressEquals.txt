public static void main(String[] args){
    System.out.println("Pressing '=' key using java.awt.Robot...");
    try{
        java.awt.Robot r=new java.awt.Robot();
        r.setAutoDelay(50);
        r.delay(300);
        r.keyPress(java.awt.event.KeyEvent.VK_EQUALS);
        r.keyRelease(java.awt.event.KeyEvent.VK_EQUALS);
        System.out.println("Done.");
    }catch(Throwable t){
        System.out.println("Failed to press key: "+t);
        t.printStackTrace();
    }
}
