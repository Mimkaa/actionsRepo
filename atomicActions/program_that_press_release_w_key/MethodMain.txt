public static void main(String[] args){System.out.println("Pressing W via java.awt.Robot...");try{java.awt.Robot r=new java.awt.Robot();r.keyPress(java.awt.event.KeyEvent.VK_W);r.delay(100);r.keyRelease(java.awt.event.KeyEvent.VK_W);System.out.println("Done.");}catch(java.awt.AWTException e){System.out.println("Robot failed: "+e);}try{new java.io.File(".ready").createNewFile();System.out.println("Created .ready file.");}catch(Exception e){System.out.println("Failed to create .ready file: "+e);}}
